varying lowp vec3 vLighting; varying lowp vec3 vColor; void main(void) { gl_FragColor = vec4(vColor*vLighting, 1.0); }
attribute vec3 aVertexPosition; attribute vec3 aVertexNormal; uniform lowp mat4 uMVCMatrix;//world uniform lowp mat4 uMVOMatrix;//object uniform lowp mat4 uPMatrix; uniform lowp mat4 uRMatrix; uniform lowp mat4 uOriMatrix; uniform lowp mat4 uScaleMatrix; uniform lowp vec3 uColor; varying lowp vec3 vLighting; varying lowp vec3 vColor; void main(void) { const lowp vec3 ambientLightColor = vec3(0.5, 0.5, 0.5); const lowp vec3 directLightDir = vec3(0.0, 0.0, 1.0); const lowp vec3 directLightColor = vec3(.7, .7, .7); gl_Position = uPMatrix * uMVCMatrix * uRMatrix * uMVOMatrix * uOriMatrix * uScaleMatrix * vec4(aVertexPosition, 1.0); lowp vec4 transformedNormal = uOriMatrix * vec4(aVertexNormal, 1.0); float intensity =max(dot(transformedNormal.xyz, directLightDir), 0.0); vLighting = ambientLightColor + (directLightColor * intensity); vColor = uColor; }
attribute vec3 aVertexPosition; uniform lowp mat4 uMVCMatrix;//world uniform lowp mat4 uMVOMatrixStart;//object uniform lowp mat4 uMVOMatrixEnd;//object uniform lowp mat4 uPMatrix; uniform lowp mat4 uRMatrix; uniform lowp vec3 uColor; varying lowp vec3 vLighting; varying lowp vec3 vColor; void main(void){ if(aVertexPosition[0] == -1.0) gl_Position = uPMatrix * uMVCMatrix * uRMatrix * uMVOMatrixStart * vec4(0.0, 0.0, 0.0, 1.0); else gl_Position = uPMatrix * uMVCMatrix * uRMatrix * uMVOMatrixEnd * vec4(0.0, 0.0, 0.0, 1.0); vLighting = vec3(1.0, 1.0, 1.0); vColor = uColor; }
